USE CASE 3: Worst Fit Memory Allocation

PROGRAM (worst_fit.c):

```c id="wf7x21"
#include <stdio.h>

int main()
{
    int nb, np, i, j;

    printf("Enter number of memory blocks: ");
    scanf("%d", &nb);

    int block[nb];
    for(i = 0; i < nb; i++)
    {
        printf("Enter size of block %d: ", i+1);
        scanf("%d", &block[i]);
    }

    printf("Enter number of processes: ");
    scanf("%d", &np);

    int process[np];
    for(i = 0; i < np; i++)
    {
        printf("Enter size of process %d: ", i+1);
        scanf("%d", &process[i]);
    }

    int allocation[np];
    for(i = 0; i < np; i++)
        allocation[i] = -1;

    for(i = 0; i < np; i++)
    {
        int worstIdx = -1;

        for(j = 0; j < nb; j++)
        {
            if(block[j] >= process[i])
            {
                if(worstIdx == -1 || block[j] > block[worstIdx])
                    worstIdx = j;
            }
        }

        if(worstIdx != -1)
        {
            allocation[i] = worstIdx;
            block[worstIdx] -= process[i];
        }
    }

    printf("\nProcess No\tProcess Size\tBlock No\n");
    for(i = 0; i < np; i++)
    {
        printf("%d\t\t%d\t\t", i+1, process[i]);
        if(allocation[i] != -1)
            printf("%d\n", allocation[i] + 1);
        else
            printf("Not Allocated\n");
    }

    return 0;
}
```
