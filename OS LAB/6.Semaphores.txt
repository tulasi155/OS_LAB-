```c id="8l2v6x"
#include <pthread.h>
#include <semaphore.h>
#include <stdio.h>

sem_t wrt;
pthread_mutex_t mutex;

int counter = 1;
int numreader = 0;

void *writer(void *wno)
{
    sem_wait(&wrt);
    counter = counter * 2;
    printf("Writer %d modified counter to %d\n", *((int *)wno), counter);
    sem_post(&wrt);
}

void *reader(void *rno)
{
    pthread_mutex_lock(&mutex);
    numreader++;
    if (numreader == 1)
        sem_wait(&wrt);
    pthread_mutex_unlock(&mutex);

    printf("Reader %d: read counter as %d\n", *((int *)rno), counter);

    pthread_mutex_lock(&mutex);
    numreader--;
    if (numreader == 0)
        sem_post(&wrt);
    pthread_mutex_unlock(&mutex);
}

int main()
{
    pthread_t read[5], write[2];
    pthread_mutex_init(&mutex, NULL);
    sem_init(&wrt, 0, 1);

    int a[5] = {1, 2, 3, 4, 5};

    for (int i = 0; i < 5; i++)
        pthread_create(&read[i], NULL, reader, &a[i]);

    for (int i = 0; i < 2; i++)
        pthread_create(&write[i], NULL, writer, &a[i]);

    for (int i = 0; i < 5; i++)
        pthread_join(read[i], NULL);

    for (int i = 0; i < 2; i++)
        pthread_join(write[i], NULL);

    pthread_mutex_destroy(&mutex);
    sem_destroy(&wrt);

    return 0;
}
```
