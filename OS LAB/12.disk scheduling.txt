PROGRAM (disk.c):

```c
#include <stdio.h>

int main()
{
    int n, i, j, head, temp, total = 0;
    int disk[50];

    printf("Enter number of disk requests: ");
    scanf("%d", &n);

    printf("Enter disk requests:\n");
    for(i = 0; i < n; i++)
        scanf("%d", &disk[i]);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    disk[n] = head;
    n++;

    // sort
    for(i = 0; i < n; i++)
    {
        for(j = i+1; j < n; j++)
        {
            if(disk[i] > disk[j])
            {
                temp = disk[i];
                disk[i] = disk[j];
                disk[j] = temp;
            }
        }
    }

    int pos;
    for(i = 0; i < n; i++)
    {
        if(disk[i] == head)
        {
            pos = i;
            break;
        }
    }

    printf("Seek Sequence: ");

    // move left
    for(i = pos; i >= 0; i--)
    {
        printf("%d -> ", disk[i]);
        if(i != pos)
            total += disk[i+1] - disk[i];
    }

    // move right
    for(i = pos+1; i < n; i++)
    {
        printf("%d -> ", disk[i]);
        total += disk[i] - disk[i-1];
    }

    printf("\nTotal Head Movement = %d\n", total);

    return 0;
}
```
