#include <stdio.h>
#include <dirent.h>

int main()
{
    DIR *d;
    struct dirent *dir;

    d = opendir(".");

    if (d)
    {
        while ((dir = readdir(d)) != NULL)
        {
            printf("%s\n", dir->d_name);
        }
        closedir(d);
    }
    else
    {
        printf("Directory not found\n");
    }

    return 0;
}

```c id="b9t0yv"
#include <sys/types.h>
#include <sys/stat.h>
#include <dirent.h>
#include <stdio.h>
#include <stdlib.h>

void listDir(char *dirName)
{
    DIR *dir;
    struct dirent *entry;
    struct stat info;
    char path[1024];

    dir = opendir(dirName);

    if (dir == NULL)
    {
        perror("Directory not found");
        exit(1);
    }

    while ((entry = readdir(dir)) != NULL)
    {
        sprintf(path, "%s/%s", dirName, entry->d_name);
        lstat(path, &info);

        if (S_ISDIR(info.st_mode))
            printf("DIR  ");
        else if (S_ISREG(info.st_mode))
            printf("FILE ");
        else if (S_ISLNK(info.st_mode))
            printf("LINK ");
        else
            printf("OTHER ");

        printf("%s\n", entry->d_name);
    }

    closedir(dir);
}

int main(int argc, char *argv[])
{
    if (argc != 2)
    {
        printf("Usage: %s <directory>\n", argv[0]);
        return 1;
    }

    listDir(argv[1]);
    return 0;
}
```
